```c
// 预编译 指在编译时包含stdio.h这个头文件
#include <stdio.h>

// 主函数 从main函数执行
int main() {
    // \n 换行符
    printf("Hello, World!\n");

    // \\ 反斜杠
    printf("Hello, World!\\");

    // \" 双引号
    printf("Hello, World!\"");

    // \t 制表符
    printf("Hello, World!\t'");

    
    return 0;
}
```
